// Программа, которая просит от пользователя ввести три числа,
// и затем печатает их сумму и произведение

#include <iostream>

using namespace std;

int main()
{
    int a, b, c;
    cout << "Vvedite poshaluyasta tri chisla\n";
    cin >> a >> b >> c;
    cout << "Summa ravna: " << a + b + с << endl;
    cout << "Proizvedeniye ravno: " << a * b * с << endl;
    return 0;
}
